1. Get the application URL by running these commands:
{{- if eq .Values.service.type "ClusterIP" }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 443:443
  Access the API at https://localhost/countries
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "bale-countries-check.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo https://$NODE_IP:$NODE_PORT/countries
{{- else if contains "LoadBalancer" .Values.service.type }}
  kubectl --namespace {{ .Release.Namespace }} get services {{ include "bale-countries-check.fullname" . }} -w
  Note: It may take a few minutes for the LoadBalancer IP to be available.
{{- end }}
